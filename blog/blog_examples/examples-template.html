class Die

attr_reader :sides

def initialize(sides)

@sides = sides

if @sides < 1

raise ArgumentError.new("Only integers greater than 0 allowed")

end

end

def roll

@roll = 1 + rand(@sides)

end

end

guess = Die.new (14)

p guess.roll
