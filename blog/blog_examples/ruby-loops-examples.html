Looping Through Ruby Arrays Example

def pad(array, min_size, value = nil) #non-destructive

additional_num = min_size - array.lengths

new_array = array.dup

while additional_num > 0 do

new_array << value

additional_num -= 1

end

new_array

end

p pad([1,5,6], 12, "index")
